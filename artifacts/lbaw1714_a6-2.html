3. Triggers

Trigger Reference TRIGGER01
Trigger Description When a community event is created, every Member of the community is invited to that event 
CREATE FUNCTION inviteAllMembers() RETURNS TRIGGER AS
$BODY$
BEGIN
  INSERT INTO 'event_member' VALUES
  (new.idEvent, member, admin)
  SELECT community_member.idmember as member, community_member.isadmin  as admin
  FROM 'community_member'
  WHERE(new.community = community_member=idcommunity);
  RETURN NEW;
END
$BODY$
LANGUAGE plpgsql;

CREATE TRIGGER eventCommunity
  AFTER INSERT ON EVENT
  FOR EACH ROW
  WHEN Event.community NOT NULL
    EXECUTE PROCEDURE inviteAllMembers();
Trigger Reference TRIGGER02
Trigger Description Before sending a Friend Request, check if the tuple or its reflex already exists 
DROP TRIGGER IF EXISTS inviteFriend;

CREATE FUNCTION checkDuplicateFriend() RETURNS TRIGGER AS
$BODY$
BEGIN
 IF EXISTS 
 (SELECT * 
    FROM "Friend"
    WHERE ("Friend".idfriend1 = NEW.idfriend1 AND "Friend".idfriend2 = NEW.idfriend2) OR
        ("Friend".idfriend1 = NEW.idfriend2 AND "Friend".idfriend2 = NEW.idfriend1))
THEN RAISE EXCEPTION 'Duplicate Friendship Tuple';
END
$BODY$
LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS inviteFriend;

CREATE TRIGGER inviteFriend
  BEFORE INSERT ON "Friend"
    EXECUTE PROCEDURE checkDuplicateFriend();
Trigger Reference TRIGGER03
Trigger Description Before buying a ticket, the available quantitiy is checked, and an invoice row is created 
CREATE FUNCTION createInvoice() RETURNS TRIGGER AS
$BODY$
BEGIN
  IF NOT EXISTS (SELECT * 
    FROM "TicketType"
    WHERE "TicketType".idTicketType = New.type AND
    "TicketType".availableQuantity - $quantity < 0)
  RAISE EXCEPTION 'Not enough tickets avaiable of that type';
  END IF;

  UPDATE TicketType
  SET availableQuantity = aavailableQuantity - $quantity
  WHERE "TicketType".idTicketType = New.type;

  INSERT INTO "invoice" VALUES
  (id, memberid, taxpayer, address, $quantity, $quantity * price, $today)
  SELECT memberid, taxpayer, address
  FROM "member"
  WHERE (member.memberid = $_SESSION['currentMember'])
  SELECT price
  FROM "tickettype"
  WHERE (new.type = tickettype.idtickettype);
  RETURN NEW;
END
$BODY$
LANGUAGE plpgsql;

CREATE TRIGGER buyTicketBefore
  BEFORE INSERT ON "ticket"
  FOR EACH ROW
    EXECUTE PROCEDURE createInvoice();
